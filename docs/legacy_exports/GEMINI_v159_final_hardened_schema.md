```sql
CREATE TABLE IF NOT EXISTS raw_mapping_cases (

      'AUTO_BOS',
      'RECLAIM',

      'NOTE'
```

### Finalized Hardened Ingestion Infrastructure Code

This script adds automated SHA-256 data hash calculations directly into your telemetry export routes. When Python requests a case dataset, it receives a cryptographic fingerprint generated from the current sequence state.

```typescript
// src/routes/rawMappingHardened.ts
import { Router, Request, Response } from 'express';
import { Database } from 'sqlite3';
import { v4 as uuidv4 } from 'uuid';
import * as crypto from 'crypto';

const router = Router();
const db = new Database(process.env.DATABASE_PATH || './dev.db');

// ==========================================
// 1. POST /api/v1/raw-mapping/cases (Case Factory)
// ==========================================
router.post('/cases', (req: Request, res: Response) => {
  const { symbol, case_name, base_timeframe, price_scale_default, notes } = req.body;
  
  if (!symbol || !case_name || !base_timeframe || !price_scale_default) {
    return res.status(400).json({ error: 'Missing core case initialization specifications' });
  }

  const case_id = uuidv4();
  const now = Date.now();

  const query = `
    INSERT INTO raw_mapping_cases (case_id, symbol, case_name, base_timeframe, price_scale_default, created_at_utc_ms)
    VALUES (?, ?, ?, ?, ?, ?);
  `;

  db.run(query, [case_id, symbol.toUpperCase(), case_name, base_timeframe, price_scale_default, now], (err) => {
    if (err) return res.status(500).json({ error: 'Failed to create case file context', details: err.message });
    return res.status(201).json({ success: true, case_id });
  });
});

// ==========================================
// 2. POST /api/v1/raw-mapping/events (Single Event Ingestion)
// ==========================================
router.post('/events', (req: Request, res: Response) => {
  const {
    event_id, case_id, symbol, timeframe, candle_time_utc_ms, candle_index,
    price, event_type, event_side, source, supersedes_event_id, notes, raw_payload_json
  } = req.body;

  const final_event_id = event_id || uuidv4();
  const now = Date.now();

  db.serialize(() => {
    db.run('BEGIN TRANSACTION;');

    // Idempotency Verification Check
    db.get('SELECT * FROM raw_mapping_events WHERE event_id = ?', [final_event_id], (err, duplicate) => {
      if (duplicate) {
        db.run('COMMIT;');
        return res.status(200).json({ success: true, duplicate: true, event: duplicate });
      }

      // Fetch Context Rule Variables from parent case container
      db.get('SELECT price_scale_default FROM raw_mapping_cases WHERE case_id = ?', [case_id], (caseErr, caseMeta: any) => {
        if (!caseMeta) {
          db.run('ROLLBACK;');
          return res.status(400).json({ error: 'Target case context does not exist on the VPS' });
        }

        const scale = caseMeta.price_scale_default;
        const price_int = price !== undefined && price !== null ? Math.round(price * scale) : null;

        db.get('SELECT COALESCE(MAX(created_order), 0) as max_order FROM raw_mapping_events WHERE case_id = ?', [case_id], (orderErr, orderMeta: any) => {
          const next_order = orderMeta.max_order + 1;

          const query = `
            INSERT INTO raw_mapping_events (
              event_id, case_id, symbol, timeframe, candle_time_utc_ms, candle_index,
              price, price_int, price_scale, event_type, event_side, source,
              created_order, schema_version, supersedes_event_id, notes, created_at_utc_ms, raw_payload_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 'raw_mapping_v1', ?, ?, ?, ?);
          `;

          db.run(query, [
            final_event_id, case_id, symbol, timeframe, candle_time_utc_ms, candle_index ?? null,
            price ?? null, price_int, scale, event_type, event_side, source || 'MANUAL',
            next_order, supersedes_event_id || null, notes || '', now,
            raw_payload_json ? JSON.stringify(raw_payload_json) : null
          ], (insErr) => {
            if (insErr) {
              db.run('ROLLBACK;');
              return res.status(500).json({ error: insErr.message });
            }
            db.run('COMMIT;', () => {
              return res.status(201).json({ success: true, event_id: final_event_id, created_order: next_order });
            });
          });
        });
      });
    });
  });
});

// ==========================================
// 3. GET /api/v1/raw-mapping/events/export (Hashed Ingestion Port)
// ==========================================
router.get('/events/export', (req: Request, res: Response) => {
  const { case_id } = req.query;
  if (!case_id) return res.status(400).json({ error: 'Missing mandatory case identifier parameter' });

  db.get('SELECT * FROM raw_mapping_cases WHERE case_id = ?', [case_id], (caseErr, caseMeta) => {
    if (!caseMeta) return res.status(404).json({ error: 'Target case context profile not found' });

    const query = `SELECT * FROM raw_mapping_events WHERE case_id = ?;`;
    db.all(query, [case_id], (err, rows: any[]) => {
      if (err) return res.status(500).json({ error: err.message });

      const sequence_by_intent = [...rows].sort((a, b) => a.created_order - b.created_order);
      const sequence_by_timeline = [...rows].sort((a, b) => a.candle_time_utc_ms - b.candle_time_utc_ms);

      // Generate a SHA-256 string fingerprint based on sorted tracking execution orders
      const fingerprintContent = sequence_by_intent.map(r => `${r.event_id}:${r.created_order}:${r.is_deleted}`).join('|');
      const ledger_hash = crypto.createHash('sha256').update(fingerprintContent).digest('hex');

      return res.status(200).json({
        meta: {
          case_id,
          schema_version: 'raw_mapping_v1',
          total_records: rows.length,
          ledger_hash
        },
        sequence_by_intent,
        sequence_by_timeline
      });
    });
  });
});

export default router;
```
